def es_primo(n):
    if n < 2:
        return False
    for i in range(2, n):
        if n % i == 0:
            return False
    return True

numeros = [2, 3, 4, 10, 13, 17, 20]
for num in numeros:
    if es_primo(num):
        print(f"{num} es primo")

def calcular_totales(diccionario_tren):
    lista_totales = []
    for vagon, animales in diccionario_tren.items():
        total = 0
        for cant in animales.values():
            total += cant
        lista_totales.append((vagon, total))
    return lista_totales

def buscar_ayudante(lista_totales):
    menor_vagon = lista_totales[0][0]
    min_cant = lista_totales[0][1]

        if total < min_cant:
            min_cant = total
            menor_vagon = vagon
    return menor_vagon

def convertir_tiempo(segundos):
    horas = segundos // 3600
    resto = segundos % 3600
    minutos = resto // 60
    segs = resto % 60
    return horas, minutes, segs

def analizar_rendimiento(lista):
    resultado = {}
    for e in lista:
        total = 0
        for nota in e["notas"]:
            total += nota
        prom = total / len(e["notas"])
        estado = "Azul" if prom >= 4.0 else "Rojo"
        resultado[e["nombre"]] = (prom, estado)
    return resultado
